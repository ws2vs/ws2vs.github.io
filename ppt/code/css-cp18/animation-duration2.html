duration数 < name数

                
.block {
    animation-name: moveToRight,translateToPink;
    animation-duration: 2s;
}

duration数 > name数

                
.block {
    animation-name: moveToRight,translateToPink;
    animation-duration: 2s,3s,10s;
}

duration有错误值时

                
.block {
    animation-name: moveToRight,translateToPink;
    animation-duration: 2s,f,10s;
}

duration有错误值时

                
.block {
    animation-name: moveToRight,translateToPink;
    animation-duration: 2s,0,10s;
}
